# Operation-program
Runs operations when given three integers and complies them to the third integer
Num1 = int (input("Enter number1 :")) 
Num2 = int (input("Enter number2 :")) 
Num3 = int (input("Enter number3 :")) 

if (Num1 + Num2== Num3):
  print (Num1, "+", Num2, "=", Num3)
if (Num1 - Num2== Num3):
  print (Num1, "-", Num2, "=", Num3)
if (Num1 * Num2== Num3):
  print (Num1, "*", Num2, "=", Num3)
if (Num1 / Num2== Num3):
  print (Num1, "/", Num2, "=", Num3)

else:
   print("None")
